.then() method

.then(onFulFilled, onRejected)

	
	var mapPromise = getGoogleMap();
	
	function showMap(map) {
	  displayMap(map);
	}
	
	function showError(map) {
	  displayError(map);
	}
	
	mapPromise.then(showMap, showError);
	

	
	var mapPromise = getGoogleMap();
	
	function showMap(map) {
	  return displayMap(map);
	}
	
	function showError(map) {
	  displayError(map);
	}
	
	mapPromise
	 .then(showMap)
	 .then(null, showError)

.then() handlers

  • Optional
  • Always async
  • Called only once
  • Catches Exceptions, and rejects with them
